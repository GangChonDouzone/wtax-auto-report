🔍 지방소득세 주요 탐지 쿼리 예시
① 지방소득세율 이탈 탐지
-- 지방소득세율 이탈 탐지: LOCTAX_AMT ÷ 원천소득세 ≠ 10%
SELECT
  l.CO_CD, l.DIV_CD, l.DOC_NO,
  'LTAX_RATE'          AS RULE_CD,
  l.LOCTAX_AMT         AS ACTUAL_VAL,
  SUM(w.INTAX_AMT) * 0.1  AS EXPECTED_VAL,
  l.LOCTAX_AMT - SUM(w.INTAX_AMT) * 0.1  AS DIFF_AMT,
  CASE WHEN ABS(l.LOCTAX_AMT - SUM(w.INTAX_AMT) * 0.1)
           / NULLIF(SUM(w.INTAX_AMT) * 0.1, 0) > 0.01
       THEN 'Y' ELSE 'N' END  AS DETECT_YN
FROM HR_TAX_LTAX_MAT l
JOIN HR_WITHREP_DTL  w ON w.CO_CD = l.CO_CD
                        AND w.DIV_CD = l.DIV_CD
                        AND LEFT(w.DOC_NO, 6) = l.RVERS_START_YM  -- 귀속기간 간접연계
WHERE l.CO_CD = ? AND l.DIV_CD = ? AND l.DOC_NO = ?
GROUP BY l.CO_CD, l.DIV_CD, l.DOC_NO, l.LOCTAX_AMT;
② 연말정산 전월이월 불일치 탐지
-- 전월 YREND_TMADRTX_AMT ≠ 당월 LMRETX_YREND_AMT
SELECT
  cur.CO_CD, cur.DIV_CD, cur.DOC_NO, cur.APLCNTAR_CD,
  'LTAX_YREND_LMREF'      AS RULE_CD,
  CASE WHEN pre.YREND_TMADRTX_AMT != cur.LMRETX_YREND_AMT
       THEN 'Y' ELSE 'N' END  AS DETECT_YN,
  cur.LMRETX_YREND_AMT    AS ACTUAL_VAL,
  pre.YREND_TMADRTX_AMT   AS EXPECTED_VAL,
  cur.LMRETX_YREND_AMT - pre.YREND_TMADRTX_AMT  AS DIFF_AMT
FROM HR_TAX_LTAX_MAT cur
JOIN HR_TAX_LTAX_MAT pre
  ON  pre.CO_CD       = cur.CO_CD
  AND pre.DIV_CD      = cur.DIV_CD
  AND pre.DOC_NO      = cur.LSMTH_DOC_NO   -- 전월 문서번호 연결
  AND pre.APLCNTAR_CD = cur.APLCNTAR_CD
WHERE cur.CO_CD = ? AND cur.DIV_CD = ? AND cur.DOC_NO = ?;
③ 인원수·과세표준 불일치 탐지
-- PAYMENT.NOPL_CNT ↔ DTL COUNT 불일치 탐지
SELECT
  p.CO_CD, p.DIV_CD, p.DOC_NO, p.APLCNTAR_CD, p.IMPR_FG,
  'LTAX_NOPL_MISMATCH'     AS RULE_CD,
  p.NOPL_CNT               AS EXPECTED_CNT,
  COUNT(d.EMP_CD)          AS ACTUAL_CNT,
  p.TAXSTD_AMT             AS EXPECTED_TAXSTD,
  SUM(d.TAXSTD_AMT)        AS ACTUAL_TAXSTD,
  CASE WHEN p.NOPL_CNT != COUNT(d.EMP_CD)
         OR ABS(p.TAXSTD_AMT - SUM(d.TAXSTD_AMT)) > 1
       THEN 'Y' ELSE 'N' END  AS DETECT_YN
FROM HR_TAX_LTAX_PAYMENT p
JOIN HR_TAX_LTAX_DTL     d
  ON  d.CO_CD       = p.CO_CD
  AND d.DIV_CD      = p.DIV_CD
  AND d.DOC_NO      = p.DOC_NO
  AND d.APLCNTAR_CD = p.APLCNTAR_CD
  AND d.IMPR_FG     = p.IMPR_FG
WHERE p.CO_CD = ? AND p.DIV_CD = ? AND p.DOC_NO = ?
GROUP BY p.CO_CD, p.DIV_CD, p.DOC_NO, p.APLCNTAR_CD, p.IMPR_FG,
         p.NOPL_CNT, p.TAXSTD_AMT;

① 소득세액 급감 탐지 (RISK_001)
-- 소득세액 급감 탐지: 전월대비 INTAX_AMT 30% 이상 감소
SELECT
  c.CO_CD, c.DIV_CD, c.DOC_NO,
  'RISK_001'            AS RULE_CD,
  CASE WHEN
    (p.INTAX_AMT - c.INTAX_AMT) / NULLIF(p.INTAX_AMT, 0) * 100 >= 30
    AND ABS(c.PAY_TOT_AMT - p.PAY_TOT_AMT) / NULLIF(p.PAY_TOT_AMT, 0) * 100 <= 10
  THEN 'Y' ELSE 'N' END  AS DETECT_YN,
  c.INTAX_AMT           AS ACTUAL_VAL,
  p.INTAX_AMT           AS EXPECTED_VAL,
  c.INTAX_AMT - p.INTAX_AMT AS DIFF_AMT,
  (c.INTAX_AMT - p.INTAX_AMT) / NULLIF(p.INTAX_AMT, 0) * 100 AS DIFF_RATE,
  NOW()                 AS ANLYZ_DT
FROM HR_WITHREP_DTL c
JOIN HR_WITHREP_MST  m  ON m.CO_CD = c.CO_CD AND m.DIV_CD = c.DIV_CD AND m.DOC_NO = c.DOC_NO
JOIN HR_WITHREP_DTL p  ON p.CO_CD = m.CO_CD AND p.DIV_CD = m.DIV_CD
                         AND p.DOC_NO = m.LSMTH_DOC_NO   -- 전월 신고서
                         AND p.SRC_LEV_FG_CD = c.SRC_LEV_FG_CD
WHERE c.CO_CD = ? AND c.DIV_CD = ? AND c.DOC_NO = ?;
② 전월 이월환급세액 불일치 탐지 (RISK_011)
-- 전월 NEMOX_REFUTX_AMT ≠ 당월 LMNREF_TAX_AMT
SELECT
  cur.CO_CD, cur.DIV_CD, cur.DOC_NO,
  'RISK_011'            AS RULE_CD,
  CASE WHEN pre.NEMOX_REFUTX_AMT != cur.LMNREF_TAX_AMT
       THEN 'Y' ELSE 'N' END  AS DETECT_YN,
  cur.LMNREF_TAX_AMT    AS ACTUAL_VAL,
  pre.NEMOX_REFUTX_AMT  AS EXPECTED_VAL,
  cur.LMNREF_TAX_AMT - pre.NEMOX_REFUTX_AMT  AS DIFF_AMT
FROM HR_WITHREP_MST cur
JOIN HR_WITHREP_MST pre ON pre.CO_CD  = cur.CO_CD
                          AND pre.DIV_CD = cur.DIV_CD
                          AND pre.DOC_NO = cur.LSMTH_DOC_NO  -- 전월 문서번호 연결
WHERE cur.CO_CD = ? AND cur.DIV_CD = ? AND cur.DOC_NO = ?;
③ 신고서 생성 후 연말정산 변경 탐지 (RISK_016)
-- 신고서 생성일시(INSERT_DT) < 연말정산 수정일시(MODIFY_DT)
SELECT
  m.CO_CD, m.DIV_CD, m.DOC_NO,
  'RISK_016'        AS RULE_CD,
  CASE WHEN MAX(y.MODIFY_DT) > m.INSERT_DT THEN 'Y' ELSE 'N' END  AS DETECT_YN,
  UNIX_TIMESTAMP(MAX(y.MODIFY_DT)) AS ACTUAL_VAL,
  UNIX_TIMESTAMP(m.INSERT_DT)       AS EXPECTED_VAL
FROM HR_WITHREP_MST m
JOIN HR_YREND_MST   y ON y.CO_CD = m.CO_CD AND y.DIV_CD = m.DIV_CD
WHERE m.CO_CD = ? AND m.DIV_CD = ? AND m.DOC_NO = ?
  AND m.YREND_YN = 'Y'
GROUP BY m.CO_CD, m.DIV_CD, m.DOC_NO, m.INSERT_DT;
